def exec_cmd(cmd)
  local_id = @local_id_counter += 1
  response = ADB::Message::Open.new(local_id, "shell:"+cmd).send_recv(@sock)
  ADB::Message::Close.new(local_id, response.arg0).send_recv(@sock)
end
